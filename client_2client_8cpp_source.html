97 auto client::send(uint32_t value, const pubkey_t& payee)
98 -> std::pair<std::optional<transaction::full_tx>,
99 std::optional<cbdc::sentinel::response>> {
100 static constexpr auto null_return
101 = std::make_pair(std::nullopt, std::nullopt);
102
103 auto spend_tx = create_transaction(value, payee);
104 if(!spend_tx.has_value()) {
105 m_logger->error("Failed to generate wallet spend tx.");
106 return null_return;
107 }
108
109 auto res = send_transaction(spend_tx.value());
110 if(!res.has_value()) {
111 return null_return;
112 }
113
114 return std::make_pair(spend_tx.value(), res.value());
115 }

139 auto client::send_transaction(const transaction::full_tx& tx)
140 -> std::optional<cbdc::sentinel::response> {
141 import_transaction(tx);
142
143 auto res = m_sentinel_client.execute_transaction(tx);
144 if(!res.has_value()) {
145 m_logger->error("Failed to send transaction to sentinel.");
146 return std::nullopt;
147 }
148
149 m_logger->info("Sentinel responded:",
150 cbdc::sentinel::to_string(res.value().m_tx_status),
151 "for",
152 to_string(transaction::tx_id(tx)));
153
154 if(res.value().m_tx_status == sentinel::tx_status::confirmed) {
155 confirm_transaction(transaction::tx_id(tx));
156 }
157
158 return res;
159 }

199 auto client::check_pending(const transaction::input& inp) -> bool {
200 // TODO: This can probably be done more efficiently, but we
201 // might have to keep a secondary map to index the pending tx set on
202 // input hash?
203 for(auto& it : m_pending_txs) {
204 for(auto& in : it.second.m_inputs) {
205 if(in == inp) {
206 return true;
207 }
208 }
209 }
210 return false;
211 }

213 auto client::abandon_transaction(const hash_t& tx_id) -> bool {
214 bool success{false};
215 const auto it = m_pending_txs.find(tx_id);
216 if(it != m_pending_txs.end()) {
217 auto tx = it->second;
218 m_pending_txs.erase(it);
219
220 // Add the used inputs back to the wallet if they are still
221 // pending and not used in any other pending transaction.
222 for(auto& i : tx.m_inputs) {
223 const auto ps_it = m_pending_spend.find(i.hash());
224 if(ps_it != m_pending_spend.end()) {
225 auto pending = check_pending(i);
226 if(!pending) {
227 m_pending_spend.erase(ps_it);
228 m_wallet.confirm_inputs({i});
229 }
230 }
231 }
232 success = true;
233 }
234
235 save();
236
237 return success;
238 }

240 auto client::confirm_transaction(const hash_t& tx_id) -> bool {
241 // TODO: Should abandon and confirm be combined somehow?
242 // for instance: finish_transaction(hash_t tx_id, bool succeeded)
243 bool success{false};
244 const auto it = m_pending_txs.find(tx_id);
245 if(it != m_pending_txs.end()) {
246 m_wallet.confirm_transaction(it->second);
247 for(auto& i : it->second.m_inputs) {
248 const auto ps_it = m_pending_spend.find(i.hash());
249 if(ps_it != m_pending_spend.end()) {
250 m_pending_spend.erase(ps_it);
251 }
252 }
253 m_pending_txs.erase(it);
254 success = true;
255 }
256
257 const auto pi_it = m_pending_inputs.find(tx_id);
258 if(pi_it != m_pending_inputs.end()) {
259 m_wallet.confirm_inputs({pi_it->second});
260 m_pending_inputs.erase(pi_it);
261 success = true;
262 }
263
264 save();
265
266 return success;
267 }

269 void client::load_client_state() {
270 std::ifstream client_file(m_client_file,
271 std::ios::binary | std::ios::in);
272 if(client_file.good()) {
273 auto deser = cbdc::istream_serializer(client_file);
274 if(!(deser >> m_pending_txs >> m_pending_inputs
275 >> m_pending_spend)) {
276 m_logger->fatal("Error deserializing client file");
277 }
278 } else {
279 m_logger->warn("Existing client file not found");
280 }
281 }

283 void client::save_client_state() {
284 std::ofstream client_file(m_client_file,
285 std::ios::binary | std::ios::trunc
286 | std::ios::out);
287 if(!client_file.good()) {
288 m_logger->fatal("Failed to open client file for saving");
289 }
290 auto ser = cbdc::ostream_serializer(client_file);
291 if(!(ser << m_pending_txs << m_pending_inputs << m_pending_spend)) {
292 m_logger->fatal("Failed to write client data");
293 }
294 }
