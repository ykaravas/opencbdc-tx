61 auto controller::init() -> bool {
62 // Thread to handle starting and stopping the message handler and dtx
63 // batch processing threads when triggered by the raft callback
64 // becoming leader or follower
65 m_start_thread = std::thread([&] {
66 start_stop_func();
67 });
68
69 // Initialize NuRaft with the state machine we just created. Register
70 // our callback function to notify us when we become a leader or
71 // follower.
72 if(!m_raft_serv.init(m_raft_params)) {
73 return false;
74 }
75
76 // Connect to the other raft nodes in our coordinator cluster
77 return m_raft_serv.build_cluster(
78 m_opts.m_coordinator_raft_endpoints[m_coordinator_id]);
79 }

81 auto controller::raft_callback(nuraft::cb_func::Type type,
82 nuraft::cb_func::Param* /* param */)
83 -> nuraft::cb_func::ReturnCode {
84 if(type == nuraft::cb_func::BecomeLeader) {
85 // We're now the leader. Inform the start/stop thread that it
86 // should start up the handler threads and initiate dtx recovery.
87 // We do this via flags and a condition variable with the actual
88 // start/stop in a separate thread to not block NuRaft internally.
89 // Since we need to use the state machine to handle recovery we
90 // need to return from this callback before we can start the
91 // process.
92 m_logger->warn("Became leader, starting coordinator");
93 {
94 std::unique_lock<std::mutex> l(m_start_mut);
95 m_start_flag = true;
96 m_stop_flag = false;
97 }
98 m_start_cv.notify_one();
99 m_logger->warn("Done with become leader handler");
100 } else if(type == nuraft::cb_func::BecomeFollower) {
101 // As above, we're now a follower. Inform the start/stop thread to
102 // kill the handler threads.
103 m_logger->warn("Became follower, stopping coordinator");
104 {
105 std::unique_lock<std::mutex> l(m_start_mut);
106 m_start_flag = false;
107 m_stop_flag = true;
108 }
109 m_start_cv.notify_one();
110 m_logger->warn("Done with become follower handler");
111 }
112 return nuraft::cb_func::ReturnCode::Ok;
113 }

115 auto
116 controller::prepare_cb(const hash_t& dtx_id,
117 const std::vector<transaction::compact_tx>& txs)
118 -> bool {
119 // Send the prepare status for this dtx ID and the txs contained within
120 // to the coordinator RSM and ensure it replicated (or failed) before
121 // returning.
122 auto comm = sm_command{{state_machine::command::prepare, dtx_id}, txs};
123 return replicate_sm_command(comm).has_value();
124 }

126 auto
127 controller::commit_cb(const hash_t& dtx_id,
128 const std::vector<bool>& complete_txs,
129 const std::vector<std::vector<uint64_t>>& tx_idxs)
130 -> bool {
131 // Send the commit status for this dtx ID and the result from prepare,
132 // along with the mapping of which txs are relevant to each shard in
133 // the prepare result to the RSM and check if it replicated.
134 auto comm = sm_command{{state_machine::command::commit, dtx_id},
135 std::make_pair(complete_txs, tx_idxs)};
136 return replicate_sm_command(comm).has_value();
137 }

151 void controller::stop() {
152 // Set the running flag to false, close the network and notify the
153 // batch trigger condition variable so the threads that check them will
154 // end and we can join the threads below.
155 {
156 std::lock_guard<std::mutex> l(m_batch_mut);
157 m_running = false;
158 }
159 m_rpc_server.reset();
160 m_batch_cv.notify_one();
161
162 // Stop each of the locking shard clients to cancel any pending RPCs
163 // and unblock any of the current dtxs so they can mark themselves as
164 // failed and stop executing.
165 {
166 // The lock just protects m_shards which we're not modifying.
167 std::shared_lock<std::shared_mutex> l(m_shards_mut);
168 for(auto& s : m_shards) {
169 s->stop();
170 }
171 }
172
173 // Join the handler and batch execution threads
174 if(m_batch_exec_thread.joinable()) {
175 m_batch_exec_thread.join();
176 }
177
178 // Join any existing dtxs still executing
179 join_execs();
180
181 // Disconnect from the shards
182 {
183 std::unique_lock<std::shared_mutex> l(m_shards_mut);
184 m_shards.clear();
185 }
186 }

188 auto controller::recovery_func() -> bool {
189 // Grab any non-completed dtxs from the state machine that were not
190 // done when the previous leader failed.
191 auto comm = sm_command{{state_machine::command::get}};
192 m_logger->info("Waiting for get SM command response");
193 auto r = replicate_sm_command(comm);
194 if(!r) {
195 // We likely stopped being the leader so we couldn't get the dtxs
196 return false;
197 }
198
199 m_logger->info("Started recovery process");
200
201 const auto& res = *r;
202 if(!res) {
203 // I don't think this should happen in practice. It might be wise
204 // to make this fatal.
205 m_logger->error("Empty response object");
206 return false;
207 }
208
209 // List of coordinators/dtxs we're going to recover
210 auto coordinators = std::vector<std::shared_ptr<distributed_tx>>();
211
212 // Deserialize the coordinator state we just retrieved from the RSM
213 auto state = coordinator_state();
214 auto deser = nuraft_serializer(*res);
215 deser >> state;
216
217 for(const auto& prep : state.m_prepare_txs) {
218 // Create a coordinator for the prepare dtx to recover
219 auto coord = std::shared_ptr<distributed_tx>();
220 {
221 std::shared_lock<std::shared_mutex> l(m_shards_mut);
222 coord = std::make_shared<distributed_tx>(prep.first,
223 m_shards,
224 m_logger);
225 }
226 // Tell the coordinator this dtx is in the prepare phase and
227 // provide the list of transactions
228 coord->recover_prepare(prep.second);
229 coordinators.emplace_back(std::move(coord));
230 }
231
232 for(const auto& com : state.m_commit_txs) {
233 auto coord = std::shared_ptr<distributed_tx>();
234 {
235 std::shared_lock<std::shared_mutex> l(m_shards_mut);
236 coord = std::make_shared<distributed_tx>(com.first,
237 m_shards,
238 m_logger);
239 }
240 // Tell the coordinator this dtx is in the commit phase and provide
241 // the flags for which dtxs to complete and the map between shards
242 // and transactions in the batch
243 coord->recover_commit(com.second.first, com.second.second);
244 coordinators.emplace_back(std::move(coord));
245 }
246
247 for(const auto& dis : state.m_discard_txs) {
248 auto coord = std::shared_ptr<distributed_tx>();
249 {
250 std::shared_lock<std::shared_mutex> l(m_shards_mut);
251 coord = std::make_shared<distributed_tx>(dis,
252 m_shards,
253 m_logger);
254 }
255 // Tell the coordinator this dtx is in the discard phase
256 coord->recover_discard();
257 coordinators.emplace_back(std::move(coord));
258 }
259
260 // Flag in case one of the dtxs fails. This would happen if we stopped
261 // being the leader mid-execution.
262 auto success = std::atomic_bool{true};
263 for(auto&& coord : coordinators) {
264 // Register the callbacks for the RSM so we track dtx state during
265 // execution
266 batch_set_cbs(*coord);
267 auto dtx_id_str = to_string(coord->get_id());
268 m_logger->info("Recovering dtx", dtx_id_str);
269 // Create a lambda that handles the execution and cleanup of the
270 // dtx
271 auto f = [&, c{std::move(coord)}, s{std::move(dtx_id_str)}](
272 size_t thread_idx) {
273 // Execute the dtx from its most recent phase
274 auto exec_res = c->execute();
275 if(!exec_res) {
276 m_logger->error("Failed to recover dtx", s);
277 // We probably stopped being the leader so set the success
278 // flag
279 success = false;
280 } else {
281 m_logger->info("Recovered dtx", s);
282 }
283 // Mark the thread we were using as done so it can be re-used
284 {
285 std::shared_lock<std::shared_mutex> l(m_exec_mut);
286 m_exec_threads[thread_idx].second = false;
287 }
288 };
289 // Schedule the lambda on an available executor thread. Blocks
290 // until there's a thread available
291 schedule_exec(std::move(f));
292 }
293
294 // Make sure we recovered fully before returning
295 join_execs();
296
297 return success;
298 }

300 void controller::batch_set_cbs(distributed_tx& c) {
301 auto s = c.get_state();
302 // Register all the RSM callbacks so the state machine tracks the state
303 // of each outstanding dtxn. Don't register the callback for the phase
304 // the dtxn is currently in as the state machine already knows about
305 // it, we can skip re-notification.
306 if(s != distributed_tx::dtx_state::prepare) {
307 c.set_prepare_cb([&](auto&& dtx_id, auto&& txs) {
308 return prepare_cb(std::forward<decltype(dtx_id)>(dtx_id),
309 std::forward<decltype(txs)>(txs));
310 });
311 }
312 if(s != distributed_tx::dtx_state::commit) {
313 c.set_commit_cb(
314 [&](auto&& dtx_id, auto&& complete_txs, auto&& tx_idxs) {
315 return commit_cb(
316 std::forward<decltype(dtx_id)>(dtx_id),
317 std::forward<decltype(complete_txs)>(complete_txs),
318 std::forward<decltype(tx_idxs)>(tx_idxs));
319 });
320 }
321 if(s != distributed_tx::dtx_state::discard) {
322 c.set_discard_cb([&](auto&& dtx_id) {
323 return discard_cb(std::forward<decltype(dtx_id)>(dtx_id));
324 });
325 }
326 if(s != distributed_tx::dtx_state::done) {
327 c.set_done_cb([&](auto&& dtx_id) {
328 return done_cb(std::forward<decltype(dtx_id)>(dtx_id));
329 });
330 }
331 }

333 void controller::batch_executor_func() {
334 while(m_running) {
335 {
336 // Wait until there are transactions ready to be processed in a
337 // dtx batch
338 std::unique_lock<std::mutex> l(m_batch_mut);
339 m_batch_cv.wait(l, [&]() {
340 return !m_current_txs->empty() || !m_running;
341 });
342 }
343 if(!m_running) {
344 break;
345 }
346
347 // Placeholders where we're going to move the current batch and map
348 // of tx to sentinel so we can send the responses.
349 auto batch = std::shared_ptr<distributed_tx>();
350 auto txs
351 = std::shared_ptr<decltype(m_current_txs)::element_type>();
352
353 // New batch we're going to swap out with the current batch being
354 // built by the handler thread
355 auto new_batch = std::shared_ptr<distributed_tx>();
356 {
357 std::shared_lock<std::shared_mutex> l(m_shards_mut);
358 new_batch
359 = std::make_shared<distributed_tx>(m_rnd.random_hash(),
360 m_shards,
361 m_logger);
362 }
363
364 // Atomically swap the current batch and tx->sentinel map with new
365 // ones so we can run this batch while the handler thread builds a
366 // new one
367 {
368 std::lock_guard<std::mutex> l(m_batch_mut);
369 batch = std::move(m_current_batch);
370 txs = std::move(m_current_txs);
371 m_current_batch = std::move(new_batch);
372 batch_set_cbs(*m_current_batch);
373 m_current_txs = std::make_shared<
374 decltype(m_current_txs)::element_type>();
375 }
376
377 // Notify the handler thread it can re-start adding transactions to
378 // the current batch
379 m_batch_cv.notify_one();
380
381 // Lambda to execute the batch and respond to the sentinel with the
382 // result
383 auto f = [&, b{std::move(batch)}, t{std::move(txs)}](
384 size_t thread_idx) {
385 auto dtxid = to_string(b->get_id());
386 m_logger->info("dtxn start:", dtxid, "size:", t->size());
387 auto s = std::chrono::high_resolution_clock::now();
388 // Execute the batch from the start
389 auto res = b->execute();
390 // For each tx result in the batch create a message with
391 // the txid and the result, and send it to the appropriate
392 // sentinel.
393 for(const auto& [tx_id, metadata] : *t) {
394 const auto& [cb_func, batch_idx] = metadata;
395 auto tx_res = std::optional<bool>();
396 if(res.has_value()) {
397 tx_res = static_cast<bool>((*res)[batch_idx]);
398 }
399 cb_func(tx_res);
400 }
401 if(!res) {
402 // We probably stopped being the leader and we don't know
403 // the result of the txs so we can't respond to the
404 // sentinels. Just warn and clean up. The new leader will
405 // recover the dtx.
406 m_logger->warn("dtxn failed:", dtxid);
407 } else {
408 auto e = std::chrono::high_resolution_clock::now();
409 auto l = (e - s).count();
410 m_logger->info("dtxn done:",
411 dtxid,
412 "t:",
413 l,
414 "size:",
415 res->size());
416 }
417 // Mark our thread as done so it can be re-used
418 {
419 std::shared_lock<std::shared_mutex> l(m_exec_mut);
420 m_exec_threads[thread_idx].second = false;
421 }
422 };
423 // Schedule our executor lambda, block until there's a thread
424 // available
425 schedule_exec(std::move(f));
426 }
427 }

429 auto controller::replicate_sm_command(const sm_command& c)
430 -> std::optional<nuraft::ptr<nuraft::buffer>> {
431 auto buf = nuraft::buffer::alloc(serialized_size(c));
432 auto ser = nuraft_serializer(*buf);
433 ser << c;
434 // Sanity check to ensure total_sz was correct
435 assert(ser.end_of_buffer());
436 // Use synchronous mode to block until replication or failure
437 return m_raft_serv.replicate_sync(buf);
438 }

440 void controller::connect_shards() {
441 // Make a network::network for each shard cluster and a locking
442 // shard client to manage RPCs. Add the clients to the m_shards map so
443 // the dtxs can use them.
444 for(size_t i{0}; i < m_shard_endpoints.size(); i++) {
445 m_logger->warn("Connecting to shard cluster", std::to_string(i));
446 auto s = std::make_shared<locking_shard::rpc::client>(
447 m_shard_endpoints[i],
448 m_shard_ranges[i],
449 *m_logger);
450 if(!s->init()) {
451 m_logger->fatal("Failed to initialize shard client");
452 }
453 {
454 std::unique_lock<std::shared_mutex> l(m_shards_mut);
455 m_shards.emplace_back(std::move(s));
456 }
457 }
458 }

460 void controller::schedule_exec(std::function<void(size_t)>&& f) {
461 // Loop until we successfully scheduled the given lambda on a thread
462 bool found_thread{false};
463 while(!found_thread) {
464 {
465 std::unique_lock<std::shared_mutex> l(m_exec_mut);
466 for(size_t i{0}; i < m_exec_threads.size(); i++) {
467 auto& thr = m_exec_threads[i];
468 // If the thread is marked done we can use it
469 if(!thr.second) {
470 // Make sure the previous thread is joined
471 if(thr.first && thr.first->joinable()) {
472 thr.first->join();
473 }
474 // Mark the thread as in-use
475 thr.second = true;
476 // Start the thread with the given lambda and provide
477 // index of the thread so it can mark itself as done
478 // Not use-after-move because the outer loop exits once
479 // f has been moved due to found_thread.
480 thr.first // NOLINTNEXTLINE(bugprone-use-after-move)
481 = std::make_shared<std::thread>(std::move(f), i);
482 found_thread = true;
483 break;
484 }
485 }
486 }
487 if(!found_thread) {
488 // For now just yield to the scheduler if there were no
489 // complete threads this time. In the future this could be a
490 // condition variable instead.
491 std::this_thread::yield();
492 }
493 }
494 }
495
496 void controller::join_execs() {
497 std::shared_lock<std::shared_mutex> l(m_exec_mut);
498 for(auto& t : m_exec_threads) {
499 if(t.first && t.first->joinable()) {
500 t.first->join();
501 }
502 }
503 }
504
505 void controller::start_stop_func() {
506 while(!m_quit) {
507 bool stopping{false};
508 bool quitting{false};
509 {
510 // Wait until we're stopping, starting or quitting
511 std::unique_lock<std::mutex> l(m_start_mut);
512 m_start_cv.wait(l, [&]() {
513 return m_start_flag || m_quit || m_stop_flag;
514 });
515 // Store our plan of action so we can release the lock on these
516 // flags in case the NuRaft handler needs to set them
517 // differently while we're busy starting/stopping.
518 if(m_quit) {
519 quitting = true;
520 stopping = true;
521 } else {
522 // Sanity check: we should be stopping or starting, not
523 // both
524 assert(m_start_flag ^ m_stop_flag);
525 if(m_stop_flag) {
526 stopping = true;
527 }
528 m_start_flag = false;
529 m_stop_flag = false;
530 }
531 }
532
533 if(stopping) {
534 m_logger->warn("Stopping coordinator");
535 stop();
536 m_logger->warn("Stopped coordinator");
537 if(quitting) {
538 m_logger->warn("Quitting");
539 break;
540 }
541 } else {
542 m_logger->warn("Stopping coordinator before start");
543 // Make sure the coordinator is stopped before starting it
544 // again to satisfy any preconditions and not leave the
545 // coordinator in a partial state
546 stop();
547 m_logger->warn("Starting coordinator");
548 start();
549 m_logger->warn("Started coordinator");
550 }
551 }
552 }

554 void controller::start() {
555 // Set the running flag to true so when we start the threads they won't
556 // immediately exit
557 {
558 std::lock_guard<std::mutex> l(m_batch_mut);
559 m_running = true;
560 }
561 m_logger->warn("Resetting sentinel network handler");
562 // Reset the handler network instance so we can re-use it
563 m_rpc_server.reset();
564 m_logger->warn("Connecting to shards");
565 // Connect to the shard clusters
566 connect_shards();
567 m_logger->warn("Became leader, recovering dtxs");
568 // Attempt recovery of existing dtxs until we stop being the leader or
569 // recovery succeeds
570 bool recovered{false};
571 do {
572 auto res = recovery_func();
573 if(!res) {
574 m_logger->error("Failed to recover, likely stopped "
575 "being leader");
576 continue;
577 }
578 recovered = true;
579 } while(!recovered && m_raft_serv.is_leader());
580 m_logger->info("Recovery complete");
581
582 // If we stopped being the leader while attempting to recover we
583 // shouldn't bother starting and handler threads
584 if(!m_raft_serv.is_leader()) {
585 return;
586 }
587
588 // Create a fresh batch to add transactions to
589 auto batch = std::shared_ptr<distributed_tx>();
590 {
591 std::shared_lock<std::shared_mutex> ll(m_shards_mut);
592 batch = std::make_shared<distributed_tx>(m_rnd.random_hash(),
593 m_shards,
594 m_logger);
595 }
596 // Register the RSM callbacks with the batch
597 batch_set_cbs(*batch);
598
599 // Atomically set the current batch and a new tx->sentinel map
600 {
601 std::lock_guard<std::mutex> ll(m_batch_mut);
602 m_current_batch = std::move(batch);
603 m_current_txs
604 = std::make_shared<decltype(m_current_txs)::element_type>();
605 }
606
607 // Start the batch executor thread
608 m_batch_exec_thread = std::thread([&] {
609 batch_executor_func();
610 });
611
612 // Listen on the coordinator endpoint and start handling incoming txs
613 auto rpc_server = std::make_unique<cbdc::rpc::tcp_server<
614 cbdc::rpc::async_server<rpc::request, rpc::response>>>(
615 m_handler_endpoint);
616 if(!rpc_server->init()) {
617 m_logger->fatal("Failed to start RPC server");
618 }
619
620 m_rpc_server = std::make_unique<decltype(m_rpc_server)::element_type>(
621 this,
622 std::move(rpc_server));
623 }
624
625 void controller::quit() {
626 // Notify the start/stop thread that we're quitting. One thread handles
627 // starting and stopping to ensure only one thing is happening at a
628 // time avoiding races on handler threads.
629 {
630 std::unique_lock<std::mutex> l(m_start_mut);
631 m_quit = true;
632 }
633 m_start_cv.notify_one();
634 if(m_start_thread.joinable()) {
635 m_start_thread.join();
636 }
637 }

652
653 auto controller::execute_transaction(transaction::compact_tx tx,
654 callback_type result_callback)
655 -> bool {
656 // If we're not the leader we can't process txs
657 if(!m_raft_serv.is_leader()) {
658 return false;
659 }
660
661 auto added = [&]() {
662 // Wait until there's space in the current batch
663 std::unique_lock<std::mutex> l(m_batch_mut);
664 m_batch_cv.wait(l, [&]() {
665 return m_current_txs->size() < m_batch_size || !m_running;
666 });
667 if(!m_running) {
668 return false;
669 }
670
671 // Make sure the TX is not already in the current batch
672 if(m_current_txs->find(tx.m_id) != m_current_txs->end()) {
673 return false;
674 }
675 // Add the tx to the current dtx batch and record its index
676 auto idx = m_current_batch->add_tx(tx);
677 // Map the index of the tx to the transaction ID and sentinel
678 // ID
679 m_current_txs->emplace(
680 tx.m_id,
681 std::make_pair(std::move(result_callback), idx));
682 return true;
