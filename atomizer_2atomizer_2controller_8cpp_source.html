55 auto controller::init() -> bool {
56 if(!m_watchtower_network.cluster_connect(
57 m_opts.m_watchtower_internal_endpoints)) {
58 m_logger->error("Failed to connect to watchtowers.");
59 return false;
60 }
61
62 auto raft_params = nuraft::raft_params();
63 raft_params.election_timeout_lower_bound_
64 = static_cast<int>(m_opts.m_election_timeout_lower);
65 raft_params.election_timeout_upper_bound_
66 = static_cast<int>(m_opts.m_election_timeout_upper);
67 raft_params.heart_beat_interval_
68 = static_cast<int>(m_opts.m_heartbeat);
69 raft_params.snapshot_distance_
70 = static_cast<int>(m_opts.m_snapshot_distance);
71 raft_params.max_append_size_
72 = static_cast<int>(m_opts.m_raft_max_batch);
73
74 if(!m_raft_node.init(raft_params)) {
75 return false;
76 }
77
78 auto raft_endpoints = std::vector<network::endpoint_t>();
79 for(const auto& s : m_opts.m_atomizer_raft_endpoints) {
80 raft_endpoints.push_back(*s);
81 }
82 if(!m_raft_node.build_cluster(raft_endpoints)) {
83 return false;
84 }
85
86 if(m_opts.m_batch_size > 1) {
87 m_tx_notify_thread = std::thread{[&] {
88 tx_notify_handler();
89 }};
90 }
91
92 m_main_thread = std::thread{[&] {
93 main_handler();
94 }};
95
96 return true;
97 }

99 auto controller::server_handler(cbdc::network::message_t&& pkt)
100 -> std::optional<cbdc::buffer> {
101 if(!m_raft_node.is_leader()) {
102 return std::nullopt;
103 }
104
105 auto maybe_req = from_buffer<request>(*pkt.m_pkt);
106 if(!maybe_req.has_value()) {
107 m_logger->error("Invalid request packet");
108 return std::nullopt;
109 }
110
111 std::visit(
112 overloaded{
113 [&](tx_notify_request& notif) {
114 m_raft_node.tx_notify(std::move(notif));
115 },
116 [&](const prune_request& p) {
117 m_raft_node.make_request(p, nullptr);
118 },
119 [&](const get_block_request& g) {
120 auto result_fn = [&, peer_id = pkt.m_peer_id](
121 raft::result_type& r,
122 nuraft::ptr<std::exception>& err) {
123 if(err) {
124 m_logger->error("Exception handling log entry:",
125 err->what());
126 return;
127 }
128
129 const auto res = r.get();
130 if(!res) {
131 m_logger->error("Requested block not found.");
132 return;
133 }
134
135 auto maybe_resp
136 = from_buffer<state_machine::response>(*res);
137 assert(maybe_resp.has_value());
138 assert(std::holds_alternative<get_block_response>(
139 maybe_resp.value()));
140 auto& resp
141 = std::get<get_block_response>(maybe_resp.value());
142 m_atomizer_network.send(resp.m_blk, peer_id);
143 };
144 m_raft_node.make_request(g, result_fn);
145 }},
146 maybe_req.value());
147
148 return std::nullopt;
149 }

151 void controller::tx_notify_handler() {
152 while(m_running) {
153 if(!m_raft_node.send_complete_txs([&](auto&& res, auto&& err) {
154 err_return_handler(std::forward<decltype(res)>(res),
155 std::forward<decltype(err)>(err));
156 })) {
157 static constexpr auto batch_send_delay
158 = std::chrono::milliseconds(20);
159 std::this_thread::sleep_for(batch_send_delay);
160 }

164 void controller::main_handler() {
165 auto last_time = std::chrono::high_resolution_clock::now();
166
167 while(m_running) {
168 const auto next_time
169 = last_time
170 + std::chrono::milliseconds(m_opts.m_target_block_interval);
171 std::this_thread::sleep_until(next_time);
172 last_time = std::chrono::high_resolution_clock::now();
173
174 if(m_raft_node.is_leader()) {
175 auto req = make_block_request();
176 auto res
177 = m_raft_node.make_request(req, [&](auto&& r, auto&& err) {
178 raft_result_handler(
179 std::forward<decltype(r)>(r),
180 std::forward<decltype(err)>(err));
181 });
182 if(!res) {
183 m_logger->error("Failed to make block at time",
184 last_time.time_since_epoch().count());
185 }
186 }
187 }

190 void controller::raft_result_handler(raft::result_type& r,
191 nuraft::ptr<std::exception>& err) {
192 if(err) {
193 return;
194 }
195
196 const auto res = r.get();
197 assert(res);
198 auto maybe_resp = from_buffer<state_machine::response>(*res);
199 assert(maybe_resp.has_value());
200 assert(
201 std::holds_alternative<make_block_response>(maybe_resp.value()));
202 auto& resp = std::get<make_block_response>(maybe_resp.value());
203
204 auto blk_pkt = make_shared_buffer(resp.m_blk);
205
206 m_atomizer_network.broadcast(blk_pkt);
207
208 m_logger->info("Block h:",
209 resp.m_blk.m_height,
210 ", nTXs:",
211 resp.m_blk.m_transactions.size(),
212 ", log idx:",
213 m_raft_node.last_log_idx(),
214 ", notifications:",
215 m_raft_node.tx_notify_count());
216
217 if(!resp.m_errs.empty()) {
218 auto buf = make_shared_buffer(resp.m_errs);
219 m_watchtower_network.broadcast(buf);
220 }
221 }

223 void controller::err_return_handler(raft::result_type& r,
224 nuraft::ptr<std::exception>& err) {
225 if(err) {
226 std::cout << "Exception handling log entry: " << err->what()
227 << std::endl;
228 return;
229 }
230
231 const auto res = r.get();
232 if(res) {
233 auto maybe_resp = from_buffer<state_machine::response>(*res);
234 assert(maybe_resp.has_value());
235 assert(std::holds_alternative<errors>(maybe_resp.value()));
236 auto& resp = std::get<errors>(maybe_resp.value());
237 m_watchtower_network.broadcast(resp);
238 }

241 auto controller::raft_callback(nuraft::cb_func::Type type,
242 nuraft::cb_func::Param* /* param */)
243 -> nuraft::cb_func::ReturnCode {
244 if(type == nuraft::cb_func::Type::BecomeFollower) {
245 // We became a follower so shutdown the client network handler and
246 // stop listening.
247 m_atomizer_network.close();
248 if(m_atomizer_server.joinable()) {
249 m_atomizer_server.join();
250 }
251 } else if(type == nuraft::cb_func::Type::BecomeLeader) {
252 // We became the leader. Ensure the previous handler thread is
253 // stopped and network shut down.
254 m_atomizer_network.close();
255 if(m_atomizer_server.joinable()) {
256 m_atomizer_server.join();
257 }
258 // Reset the client network so we can use it again.
259 m_atomizer_network.reset();
260 // Start listening on our client endpoint and start the handler
261 // thread.
262 auto as = m_atomizer_network.start_server(
263 m_opts.m_atomizer_endpoints[m_atomizer_id],
264 [&](auto&& pkt) {
265 return server_handler(std::forward<decltype(pkt)>(pkt));
266 });
267
268 if(!as.has_value()) {
269 m_logger->fatal("Failed to establish atomizer server.");
270 }
271 m_atomizer_server = std::move(as.value());
272 }
273 return nuraft::cb_func::ReturnCode::Ok;
274 }
