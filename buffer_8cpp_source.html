77 auto buffer::from_hex(const std::string& hex) -> std::optional<buffer> {
78 constexpr auto max_size = 102400;
79 if(hex.empty() || ((hex.size() % 2) != 0) || (hex.size() > max_size)) {
80 return std::nullopt;
81 }
82
83 auto ret = cbdc::buffer();
84
85 for(size_t i = 0; i < hex.size(); i += 2) {
86 unsigned int v{};
87 std::stringstream s;
88 // TODO: This function is likely very slow. At some point this
89 // could be converted to lookup table that we make ourselves.
90 s << std::hex << hex.substr(i, 2);
91 if(!(s >> v)) {
92 return std::nullopt;
93 }
94 ret.m_data.push_back(static_cast<std::byte>(v));
95 }
96
97 return ret;
98 }
