42 auto state_machine::commit(nuraft::ulong log_idx, nuraft::buffer& data)
43 -> nuraft::ptr<nuraft::buffer> {
44 m_last_committed_idx = log_idx;
45 auto req = from_buffer<request>(data);
46 assert(req.has_value());
47
48 auto resp = std::visit(
49 overloaded{
50 [&](aggregate_tx_notify_request& r)
51 -> std::optional<response> {
52 auto errs = errors();
53 for(auto&& msg : r.m_agg_txs) {
54 auto err = m_atomizer->insert_complete(
55 msg.m_oldest_attestation,
56 std::move(msg.m_tx));
57
58 if(err.has_value()) {
59 errs.push_back(*err);
60 }
61 m_tx_notify_count++;
62 }
63
64 if(!errs.empty()) {
65 return errs;
66 }
67
68 return std::nullopt;
69 },
70 [&](const make_block_request& /* r */)
71 -> std::optional<response> {
72 auto [blk, errs] = m_atomizer->make_block();
73 m_blocks->emplace(blk.m_height, blk);
74 return make_block_response{blk, errs};
75 },
76 [&](const get_block_request& r) -> std::optional<response> {
77 auto it = m_blocks->find(r.m_block_height);
78 if(it != m_blocks->end()) {
79 return get_block_response{it->second};
80 }
81 return std::nullopt;
82 },
83 [&](const prune_request& r) -> std::optional<response> {
84 for(auto it = m_blocks->begin(); it != m_blocks->end();) {
85 if(it->second.m_height < r.m_block_height) {
86 it = m_blocks->erase(it);
87 } else {
88 it++;
89 }
90 }
91 return std::nullopt;
92 },
93 },
94 req.value());
95 if(!resp.has_value()) {
96 return nullptr;
97 }
98 return make_buffer<response, nuraft::ptr<nuraft::buffer>>(
99 resp.value());
100 }

102 auto
103 state_machine::read_logical_snp_obj(nuraft::snapshot& s,
104 void*& /* user_snp_ctx */,
105 nuraft::ulong /* obj_id */,
106 nuraft::ptr<nuraft::buffer>& data_out,
107 bool& is_last_obj) -> int {
108 auto path = get_snapshot_path(s.get_last_log_idx());
109 {
110 std::shared_lock<std::shared_mutex> l(m_snp_mut);
111 auto ss = std::ifstream(path, std::ios::in | std::ios::binary);
112 if(!ss.good()) {
113 // Requested snapshot doesn't exit anymore, not fatal
114 return -1;
115 }
116 auto err = std::error_code();
117 auto sz = std::filesystem::file_size(path, err);
118 if(err) {
119 // If we got this far, this should work unless our system is
120 // broken
121 std::exit(EXIT_FAILURE);
122 }
123 auto buf = nuraft::buffer::alloc(sz);
124 auto read_vec = std::vector<char>(sz);
125 ss.read(read_vec.data(),
126 static_cast<std::streamsize>(buf->size()));
127 if(!ss.good()) {
128 // If we got this far, this should work unless our system is
129 // broken
130 std::exit(EXIT_FAILURE);
131 }
132 std::memcpy(buf->data_begin(), read_vec.data(), sz);
133 data_out = std::move(buf);
134 }
135
136 // TODO: send and receive in chunks
137 is_last_obj = true;
138
139 return 0;
140 }

142 void state_machine::save_logical_snp_obj(nuraft::snapshot& s,
143 nuraft::ulong& obj_id,
144 nuraft::buffer& data,
145 bool /* is_first_obj */,
146 bool /* is_last_obj */) {
147 assert(obj_id == 0);
148 auto tmp_path = get_tmp_path();
149 {
150 std::unique_lock<std::shared_mutex> l(m_snp_mut);
151 auto ss = std::ofstream(tmp_path,
152 std::ios::out | std::ios::trunc
153 | std::ios::binary);
154 if(!ss.good()) {
155 // Since we're the exclusive writer, this should work
156 std::exit(EXIT_FAILURE);
157 }
158
159 auto write_vec = std::vector<char>(data.size());
160 std::memcpy(write_vec.data(), data.data_begin(), data.size());
161 ss.write(write_vec.data(),
162 static_cast<std::streamsize>(data.size()));
163 if(!ss.good()) {
164 std::exit(EXIT_FAILURE);
165 }
166
167 ss.flush();
168 ss.close();
169
170 auto path = get_snapshot_path(s.get_last_log_idx());
171 auto err = std::error_code();
172 std::filesystem::rename(tmp_path, path, err);
173 if(err) {
174 std::exit(EXIT_FAILURE);
175 }
176 }
177
178 obj_id++;
179 }

181 auto state_machine::apply_snapshot(nuraft::snapshot& s) -> bool {
182 auto snp = read_snapshot(s.get_last_log_idx());
183 if(snp) {
184 m_blocks = snp->m_blocks;
185 m_atomizer = snp->m_atomizer;
186 m_last_committed_idx = s.get_last_log_idx();
187 }
188 return snp.has_value();

203 void state_machine::create_snapshot(
204 nuraft::snapshot& s,
205 nuraft::async_result<bool>::handler_type& when_done) {
206 assert(s.get_last_log_idx() == last_commit_index());
207 nuraft::ptr<std::exception> except(nullptr);
208 bool ret = true;
209
210 auto snp_ser = s.serialize();
211 auto snp = snapshot{m_atomizer,
212 nuraft::snapshot::deserialize(*snp_ser),
213 m_blocks};
214
215 auto tmp_path = get_tmp_path();
216 auto path = get_snapshot_path(s.get_last_log_idx());
217 {
218 std::unique_lock<std::shared_mutex> l(m_snp_mut);
219 auto ss = std::ofstream(tmp_path,
220 std::ios::out | std::ios::trunc
221 | std::ios::binary);
222 if(!ss.good()) {
223 // We're the exclusive writer so these file operations should
224 // work
225 std::exit(EXIT_FAILURE);
226 }
227
228 auto ser = cbdc::ostream_serializer(ss);
229 if(!(ser << snp)) {
230 std::exit(EXIT_FAILURE);
231 }
232
233 ss.flush();
234 ss.close();
235
236 auto err = std::error_code();
237 std::filesystem::rename(tmp_path, path, err);
238 if(err) {
239 std::exit(EXIT_FAILURE);
240 }
241
242 for(const auto& p :
243 std::filesystem::directory_iterator(m_snapshot_dir)) {
244 auto name = p.path().filename().generic_string();
245 if(name == m_tmp_file
246 || std::stoull(name) < s.get_last_log_idx()) {
247 std::filesystem::remove(p, err);
248 if(err) {
249 std::exit(EXIT_FAILURE);
250 }
251 }
252 }
253 }
254
255 when_done(ret, except);
256 }

270 auto state_machine::read_snapshot(uint64_t idx)
271 -> std::optional<snapshot> {
272 std::shared_lock<std::shared_mutex> l(m_snp_mut);
273 auto open_fail_fatal = false;
274 if(idx == 0) {
275 uint64_t max_idx{0};
276 auto err = std::error_code();
277 for(const auto& p :
278 std::filesystem::directory_iterator(m_snapshot_dir, err)) {
279 if(err) {
280 std::exit(EXIT_FAILURE);
281 }
282 auto name = p.path().filename().generic_string();
283 if(name == m_tmp_file) {
284 continue;
285 }
286 auto f_idx = std::stoull(name);
287 if(f_idx > max_idx) {
288 max_idx = f_idx;
289 }
290 }
291
292 if(max_idx == 0) {
293 return std::nullopt;
294 }
295
296 idx = max_idx;
297 open_fail_fatal = true;
298 }
299
300 auto path = get_snapshot_path(idx);
301
302 auto ss = std::ifstream(path, std::ios::in | std::ios::binary);
303 if(!ss.good()) {
304 if(open_fail_fatal) {
305 std::exit(EXIT_FAILURE);
306 }
307 return std::nullopt;
308 }
309 auto err = std::error_code();
310 auto sz = std::filesystem::file_size(path, err);
311 if(err) {
312 std::exit(EXIT_FAILURE);
313 }
314 auto deser = cbdc::istream_serializer(ss);
315 auto new_atm = std::make_shared<atomizer>(0, m_stxo_cache_depth);
316 auto new_blocks = std::make_shared<decltype(m_blocks)::element_type>();
317 auto snp
318 = snapshot{std::move(new_atm), nullptr, std::move(new_blocks)};
319 if(!(deser >> snp)) {
320 std::exit(EXIT_FAILURE);
321 }
322 snp.m_snp->set_size(sz);
323 return snp;
324 }
