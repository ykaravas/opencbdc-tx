31 auto state_machine::commit(uint64_t log_idx, nuraft::buffer& data)
32 -> nuraft::ptr<nuraft::buffer> {
33 m_last_committed_idx = log_idx;
34
35 auto resp = blocking_call(data);
36 if(!resp.has_value()) {
37 // TODO: This would only happen if there was a deserialization
38 // error with the request. Maybe we should abort here as such an
39 // event would imply a bug in the coordinator.
40 return nullptr;
41 }
42
43 return resp.value();
44 }

71 auto state_machine::process_request(cbdc::locking_shard::rpc::request req)
72 -> cbdc::locking_shard::rpc::response {
73 auto dtxid_str = to_string(req.m_dtx_id);
74 return std::visit(
75 overloaded{[&](rpc::lock_params&& params)
76 -> cbdc::locking_shard::rpc::response {
77 m_logger->info("Processing lock",
78 dtxid_str,
79 "with",
80 params.size(),
81 "txs");
82 auto res = m_shard->lock_outputs(std::move(params),
83 req.m_dtx_id);
84 assert(res.has_value());
85 m_logger->info("Done lock", dtxid_str);
86 return res.value();
87 },
88 [&](rpc::apply_params&& params)
89 -> cbdc::locking_shard::rpc::response {
90 m_logger->info("Processing apply", dtxid_str);
91 [[maybe_unused]] auto res
92 = m_shard->apply_outputs(std::move(params),
93 req.m_dtx_id);
94 assert(res);
95 m_logger->info("Done apply", dtxid_str);
96 return rpc::apply_response();
97 },
98 [&](rpc::discard_params&& /* params */)
99 -> cbdc::locking_shard::rpc::response {
100 m_logger->info("Processing discard", dtxid_str);
101 [[maybe_unused]] auto res
102 = m_shard->discard_dtx(req.m_dtx_id);
103 assert(res);
104 m_logger->info("Done discard", dtxid_str);
105 return rpc::discard_response();
106 }},
107 std::move(req.m_params));
108 }
109}
