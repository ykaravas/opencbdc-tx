85 auto send_request(cbdc::buffer request_buf,

87 response_action_type response_action) -> bool {
88 {
89 std::unique_lock<std::mutex> l(m_responses_mut);

100 p.set_value(std::move(value));

102 [&](raw_callback_type& cb) {

108 auto call_raw(cbdc::buffer request_buf,
109 request_id_type request_id,
110 std::chrono::milliseconds timeout)
111 -> std::optional<response_type> override {
112 auto response_promise = promise_type();
113 auto response_future = response_promise.get_future();
114
115 if(!send_request(std::move(request_buf),
116 request_id,
117 std::move(response_promise))) {
118 set_response(request_id, std::nullopt);
119 return std::nullopt;
120 }
121
122 if(timeout != std::chrono::milliseconds::zero()) {
123 auto res = response_future.wait_for(timeout);
124 if(res == std::future_status::timeout) {
125 set_response(request_id, std::nullopt);
126 return std::nullopt;
127 }
128 }
129
130 return response_future.get();
131 }

133 auto response_handler(network::message_t&& msg)
134 -> std::optional<buffer> {
135 auto resp
136 = client<Request, Response>::deserialize_response(*msg.m_pkt);
137 if(resp.has_value()) {
138 set_response(resp.value().m_header.m_request_id,
139 std::move(resp.value()));
140 }
141 return std::nullopt;
142 }

144 void set_response(request_id_type request_id,
145 std::optional<response_type> value) {
146 auto response_node = [&]() {
147 std::unique_lock<std::mutex> l(m_responses_mut);
148 return m_responses.extract(request_id);
149 }();
150
151 if(!response_node.empty()) {
152 set_response_value(response_node.mapped(), std::move(value));
153 }
154 }
155
156 auto call_raw(cbdc::buffer request_buf,
157 request_id_type request_id,
158 raw_callback_type response_callback) -> bool override {
159 if(!send_request(std::move(request_buf),
160 request_id,
161 std::move(response_callback))) {
162 {
163 std::unique_lock<std::mutex> l(m_responses_mut);
164 m_responses.erase(request_id);
165 }
166 return false;
167 }
168
169 return true;
170 }
