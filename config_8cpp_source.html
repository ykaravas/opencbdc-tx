13 auto parse_ip_port(const std::string& in_str) -> network::endpoint_t {
14 // TODO: error handling for string parsing
15 std::istringstream ss(in_str);
16
17 std::string host;
18 [[maybe_unused]] const auto& host_res = std::getline(ss, host, ':');
19 assert(host_res);
20
21 std::string port_str;
22 [[maybe_unused]] const auto& port_res
23 = std::getline(ss, port_str, ':');
24 assert(port_res);
25
26 auto port = std::stoul(port_str);
27 assert(port <= std::numeric_limits<unsigned short>::max());
28
29 return {host, static_cast<unsigned short>(port)};
30 }

230 auto read_shard_endpoints(options& opts, const parser& cfg)
231 -> std::optional<std::string> {
232 const auto shard_count = cfg.get_ulong(shard_count_key).value_or(0);
233 if(opts.m_twophase_mode) {
234 opts.m_locking_shard_endpoints.resize(shard_count);
235 opts.m_locking_shard_raft_endpoints.resize(shard_count);
236 opts.m_locking_shard_readonly_endpoints.resize(shard_count);
237 for(size_t i{0}; i < shard_count; i++) {
238 const auto node_count_key = get_shard_node_count_key(i);
239 const auto node_count = cfg.get_ulong(node_count_key);
240 if(!node_count) {
241 return "No node count specified for shard "
242 + std::to_string(i) + " (" + node_count_key + ")";
243 }
244 for(size_t j{0}; j < *node_count; j++) {
245 const auto raft_ep_key = get_shard_raft_endpoint_key(i, j);
246 const auto raft_ep = cfg.get_endpoint(raft_ep_key);
247 if(!raft_ep) {
248 return "No raft endpoint specified for shard "
249 + std::to_string(i) + " node " + std::to_string(j)
250 + " (" + raft_ep_key + ")";
251 }
252 opts.m_locking_shard_raft_endpoints[i].emplace_back(
253 *raft_ep);
254
255 const auto ep_key = get_shard_endpoint_key(i, j);
256 const auto ep = cfg.get_endpoint(ep_key);
257 if(!ep) {
258 return "No endpoint specified for shard "
259 + std::to_string(i) + " node "
260 + std::to_string(j);
261 }
262 opts.m_locking_shard_endpoints[i].emplace_back(*ep);
263
264 const auto ro_ep_key
265 = get_shard_readonly_endpoint_key(i, j);
266 const auto ro_ep = cfg.get_endpoint(ro_ep_key);
267 if(!ro_ep) {
268 return "No read-only endpoint specified for shard "
269 + std::to_string(i) + " node " + std::to_string(j)
270 + " (" + ro_ep_key + ")";
271 }
272 opts.m_locking_shard_readonly_endpoints[i].emplace_back(
273 *ro_ep);
274 }
275 }
276 }
277
278 return std::nullopt;
279 }

281 auto read_shard_options(options& opts, const parser& cfg)
282 -> std::optional<std::string> {
283 const auto shard_count = cfg.get_ulong(shard_count_key).value_or(0);
284 for(size_t i{0}; i < shard_count; i++) {
285 if(!opts.m_twophase_mode) {
286 const auto shard_ep_key = get_shard_endpoint_key(i);
287 const auto shard_ep = cfg.get_endpoint(shard_ep_key);
288 if(!shard_ep) {
289 return "No endpoint specified for shard "
290 + std::to_string(i) + " (" + shard_ep_key + ")";
291 }
292 opts.m_shard_endpoints.push_back(*shard_ep);
293
294 const auto shard_db_str = get_shard_db_key(i);
295 const auto shard_db = cfg.get_string(shard_db_str);
296 if(!shard_db) {
297 return "No db directory specified for shard "
298 + std::to_string(i) + " (" + shard_db_str + ")";
299 }
300 opts.m_shard_db_dirs.push_back(*shard_db);
301 }
302
303 const auto shard_loglevel_key = get_shard_loglevel_key(i);
304 const auto shard_loglevel = cfg.get_loglevel(shard_loglevel_key)
305 .value_or(defaults::log_level);
306 opts.m_shard_loglevels.push_back(shard_loglevel);
307
308 const auto start_key = get_shard_start_key(i);
309 const auto range_start = cfg.get_ulong(start_key);
310 if(!range_start) {
311 return "No range start specified for shard "
312 + std::to_string(i) + " (" + start_key + ")";
313 }
314 const auto end_key = get_shard_end_key(i);
315 const auto range_end = cfg.get_ulong(end_key);
316 if(!range_end) {
317 return "No range end specified for shard " + std::to_string(i)
318 + " (" + end_key + ")";
319 }
320 const auto shard_range
321 = std::make_pair(static_cast<uint8_t>(*range_start),
322 static_cast<uint8_t>(*range_end));
323 opts.m_shard_ranges.push_back(shard_range);
324 }
325
326 opts.m_shard_completed_txs_cache_size
327 = cfg.get_ulong(shard_completed_txs_cache_size)
328 .value_or(opts.m_shard_completed_txs_cache_size);
329
330 opts.m_seed_from = cfg.get_ulong(seed_from).value_or(opts.m_seed_from);
331 opts.m_seed_to = cfg.get_ulong(seed_to).value_or(opts.m_seed_to);
332 if(opts.m_seed_from != opts.m_seed_to) {
333 auto priv_str = cfg.get_string(seed_privkey);
334 if(!priv_str) {
335 return "Seed range defined but missing a private key";
336 }
337 if(priv_str.value().size() != sizeof(privkey_t) * 2) {
338 return "Invalid seed private key length";
339 }
340 opts.m_seed_privkey = hash_from_hex(priv_str.value());
341 opts.m_seed_value
342 = cfg.get_ulong(seed_value).value_or(opts.m_seed_value);
343 }
344
345 return std::nullopt;
346 }

577 auto read_options(const std::string& config_file)
578 -> std::variant<options, std::string> {
579 auto opts = options{};
580 auto cfg = parser(config_file);
581
582 opts.m_twophase_mode = cfg.get_ulong(two_phase_mode).value_or(0) != 0;
583
584 auto err = read_sentinel_options(opts, cfg);
585 if(err.has_value()) {
586 return err.value();
587 }
588
589 err = read_shard_endpoints(opts, cfg);
590 if(err.has_value()) {
591 return err.value();
592 }
593
594 err = read_shard_options(opts, cfg);
595 if(err.has_value()) {
596 return err.value();
597 }
598
599 err = read_atomizer_options(opts, cfg);
600 if(err.has_value()) {
601 return err.value();
602 }

638 auto check_options(const options& opts) -> std::optional<std::string> {
639 // TODO: refactor options to use maps rather than vectors and move all
640 // config checking to this function.
641 if(opts.m_twophase_mode) {
642 if(opts.m_sentinel_endpoints.empty()) {
643 return "Two-phase mode requires at least one configured "
644 "sentinel";
645 }
646 if(opts.m_locking_shard_endpoints.empty()) {
647 return "Two-phase mode requires at least one configured shard";
648 }
649 if(opts.m_coordinator_endpoints.empty()) {
650 return "Two-phase mode required at least one configured "

652 }
653 } else {
654 if(opts.m_watchtower_client_endpoints.empty()) {
655 return "Atomizer mode requires at least one configured "
656 "watchtower";
657 }
658 if(opts.m_archiver_endpoints.empty()) {
659 return "Atomizer mode requires at least one configured "
660 "archiver";
661 }
662 if(opts.m_shard_endpoints.empty()
663 && !opts.m_sentinel_endpoints.empty()) {
664 return "Sentinels require at least one configured shard";
665 }
666 if(opts.m_atomizer_endpoints.empty()) {
667 return "Atomizer mode requires at least one configured "
668 "atomizer";
669 }
670 }
671
672 if(opts.m_seed_from != opts.m_seed_to) {
673 if(opts.m_seed_from > opts.m_seed_to) {
674 return "shard_seed_from > shard_seed_to";
675 }
676 if(opts.m_seed_value == 0) {
677 return "Seed range defined but value is zero";
678 }
679 }

681 return std::nullopt;

689 auto loadgen_seed_range(const options& opts, size_t gen_id)
690 -> std::pair<size_t, size_t> {
691 assert(gen_id < opts.m_loadgen_count);
692 auto total_seed_range = opts.m_seed_to - opts.m_seed_from;
693 auto seed_range_sz = total_seed_range / opts.m_loadgen_count;
694 auto our_range_start = opts.m_seed_from + (gen_id * seed_range_sz);
695 auto our_range_end = our_range_start + seed_range_sz - 1;
696 return {our_range_start, our_range_end};

772 auto parser::find_or_env(const std::string& key) const
773 -> std::optional<value_t> {
774 auto upper_key = key;

776 upper_key.end(),

779 return std::toupper(c);
780 });
781 if(const auto* env_v = std::getenv(upper_key.c_str())) {
782 auto value = std::string(env_v);

784 }

787 if(it != m_options.end()) {
788 return it->second;
789 }

794 auto parser::parse_value(const std::string& value) -> value_t {
795 if(value[0] != '\"' && value[value.size() - 1] != '\"') {
796 if(value.find('.') == std::string::npos) {
797 const auto as_int = static_cast<size_t>(std::stoull(value));
798 return as_int;
799 }
800 const auto as_dbl = std::stod(value);
801 return as_dbl;
802 }
803
804 const auto unquoted = value.substr(1, value.size() - 2);
805 return unquoted;
806 }
