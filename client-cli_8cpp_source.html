22auto mint_command(cbdc::client& client, const std::vector<std::string>& args)
23 -> bool {
24 static constexpr auto min_mint_arg_count = 7;
25 if(args.size() < min_mint_arg_count) {
26 std::cerr << "Mint requires args <n outputs> <output value>"
27 << std::endl;
28 return false;
29 }
30
31 const auto n_outputs = std::stoull(args[5]);
32 const auto output_val = std::stoul(args[6]);
33
34 const auto mint_tx
35 = client.mint(n_outputs, static_cast<uint32_t>(output_val));
36 std::cout << cbdc::to_string(cbdc::transaction::tx_id(mint_tx))
37 << std::endl;
38 return true;
39}

41auto decode_address(const std::string& addr_str)
42 -> std::optional<cbdc::hash_t> {
43 // TODO: if/when bech32m is merged into Bitcoin Core, switch to that.
44 // see: https://github.com/bitcoin/bitcoin/pull/20861
45 // TODO: move address encoding/decoding into cbdc::client.
46 const auto [hrp, enc_data] = bech32::Decode(addr_str);
47 if(hrp != cbdc::config::bech32_hrp) {
48 std::cout << "Invalid address encoding" << std::endl;
49 return std::nullopt;
50 }
51 auto data = std::vector<uint8_t>();
52 ConvertBits<bech32_bits_per_symbol, bits_per_byte, false>(
53 [&](uint8_t c) {
54 data.push_back(c);
55 },
56 enc_data.begin(),
57 enc_data.end());
58
59 auto pubkey = cbdc::hash_t();
60 if(data[0] != static_cast<uint8_t>(cbdc::client::address_type::public_key)
61 || data.size() != pubkey.size() + 1) {
62 std::cout << "Address is not a supported type" << std::endl;
63 return std::nullopt;
64 }
65 data.erase(data.begin());
66 std::copy_n(data.begin(), pubkey.size(), pubkey.begin());
67
68 return pubkey;
69}

71void print_tx_result(const std::optional<cbdc::transaction::full_tx>& tx,
72 const std::optional<cbdc::sentinel::response>& resp,
73 const cbdc::hash_t& pubkey) {
74 std::cout << "tx_id:" << std::endl
75 << cbdc::to_string(cbdc::transaction::tx_id(tx.value()))
76 << std::endl;
77 const auto inputs = cbdc::client::export_send_inputs(tx.value(), pubkey);
78 for(const auto& inp : inputs) {
79 auto buf = cbdc::make_buffer(inp);
80 std::cout << "Data for recipient importinput:" << std::endl
81 << buf.to_hex() << std::endl;
82 }
83
84 if(resp.has_value()) {
85 std::cout << "Sentinel responded: "
86 << cbdc::sentinel::to_string(resp.value().m_tx_status)
87 << std::endl;
88 if(resp.value().m_tx_error.has_value()) {
89 std::cout << "Validation error: "
90 << cbdc::transaction::validation::to_string(
91 resp.value().m_tx_error.value())
92 << std::endl;
93 }
94 }
95}

97auto send_command(cbdc::client& client, const std::vector<std::string>& args)
98 -> bool {
99 static constexpr auto min_send_arg_count = 7;
100 if(args.size() < min_send_arg_count) {
101 std::cerr << "Send requires args <value> <pubkey>" << std::endl;
102 return false;
103 }
104
105 const auto value = std::stoul(args[5]);
106 static constexpr auto address_arg_idx = 6;
107 auto pubkey = decode_address(args[address_arg_idx]);
108 if(!pubkey.has_value()) {
109 std::cout << "Could not decode address" << std::endl;
110 return false;
111 }
112
113 const auto [tx, resp]
114 = client.send(static_cast<uint32_t>(value), pubkey.value());
115 if(!tx.has_value()) {
116 std::cout << "Could not generate valid send tx." << std::endl;
117 return false;
118 }
119
120 print_tx_result(tx, resp, pubkey.value());
121 return true;
122}

154void newaddress_command(cbdc::client& client) {
155 const auto addr = client.new_address();
156 auto addr_vec
157 = std::vector<uint8_t>(sizeof(cbdc::client::address_type::public_key)
158 + std::tuple_size<decltype(addr)>::value);
159 addr_vec[0] = static_cast<uint8_t>(cbdc::client::address_type::public_key);
160 std::copy_n(addr.begin(),
161 addr.size(),
162 addr_vec.begin()
163 + sizeof(cbdc::client::address_type::public_key));
164 auto data = std::vector<uint8_t>();
165 ConvertBits<bits_per_byte, bech32_bits_per_symbol, true>(
166 [&](uint8_t c) {
167 data.push_back(c);
168 },
169 addr_vec.begin(),
170 addr_vec.end());
171 std::cout << bech32::Encode(cbdc::config::bech32_hrp, data) << std::endl;
172}

174auto importinput_command(cbdc::client& client,
175 const std::vector<std::string>& args) -> bool {
176 static constexpr auto input_arg_idx = 5;
177 auto buffer = cbdc::buffer::from_hex(args[input_arg_idx]);
178 if(!buffer.has_value()) {
179 std::cout << "Invalid input encoding." << std::endl;
180 return false;
181 }
182
183 auto in = cbdc::from_buffer<cbdc::transaction::input>(buffer.value());
184 if(!in.has_value()) {
185 std::cout << "Invalid input" << std::endl;
186 return false;
187 }
188 client.import_send_input(in.value());
189 return true;
190}

192auto confirmtx_command(cbdc::client& client,
193 const std::vector<std::string>& args) -> bool {
194 const auto tx_id = cbdc::hash_from_hex(args[5]);
195 auto success = client.confirm_transaction(tx_id);
196 if(!success) {
197 std::cout << "Unknown TXID" << std::endl;
198 return false;
199 }
200 std::cout << "Confirmed. Balance: "
201 << cbdc::client::print_amount(client.balance())
202 << " UTXOs: " << client.utxo_count() << std::endl;
203 return true;
204}

207auto main(int argc, char** argv) -> int {
208 auto args = cbdc::config::get_args(argc, argv);
209 static constexpr auto min_arg_count = 5;
210 if(args.size() < min_arg_count) {
211 std::cerr << "Usage: " << args[0]
212 << " <config file> <client file> <wallet file> <command>"
213 << " <args...>" << std::endl;
214 return 0;
215 }
216
217 auto cfg_or_err = cbdc::config::load_options(args[1]);
218 if(std::holds_alternative<std::string>(cfg_or_err)) {
219 std::cerr << "Error loading config file: "
220 << std::get<std::string>(cfg_or_err) << std::endl;
221 return -1;
222 }
223
224 auto opts = std::get<cbdc::config::options>(cfg_or_err);
225
226 SHA256AutoDetect();
227
228 const auto wallet_file = args[3];
229 const auto client_file = args[2];
230
231 auto logger = std::make_shared<cbdc::logging::log>(
232 cbdc::config::defaults::log_level);
233
234 auto client = std::unique_ptr<cbdc::client>();
235 if(opts.m_twophase_mode) {
236 client = std::make_unique<cbdc::twophase_client>(opts,
237 logger,
238 wallet_file,
239 client_file);
240 } else {
241 client = std::make_unique<cbdc::atomizer_client>(opts,
242 logger,
243 wallet_file,
244 client_file);
245 }
246
247 if(!client->init()) {
248 return -1;
249 }
250
251 const auto command = std::string(args[4]);
252 if(command == "mint") {
253 if(!mint_command(*client, args)) {
254 return -1;
255 }
256 } else if(command == "send") {
257 if(!send_command(*client, args)) {
258 return -1;
259 }
260 } else if(command == "fan") {
261 if(!fan_command(*client, args)) {
262 return -1;
263 }
264 } else if(command == "sync") {
265 client->sync();
266 } else if(command == "newaddress") {
267 newaddress_command(*client);
268 } else if(command == "info") {
269 const auto balance = client->balance();
270 const auto n_txos = client->utxo_count();
271 std::cout << "Balance: " << cbdc::client::print_amount(balance)
272 << ", UTXOs: " << n_txos
273 << ", pending TXs: " << client->pending_tx_count()
274 << std::endl;
275 } else if(command == "importinput") {
276 if(!importinput_command(*client, args)) {
277 return -1;
278 }
279 } else if(command == "confirmtx") {
280 if(!confirmtx_command(*client, args)) {
281 return -1;
282 }
283 } else {
284 std::cerr << "Unknown command" << std::endl;
285 }
286
287 // TODO: check that the send queue has drained before closing
288 // the network. For now, just sleep.
289 static constexpr auto shutdown_delay = std::chrono::milliseconds(100);
290 std::this_thread::sleep_for(shutdown_delay);
291
292 return 0;
293}
