41 auto atomizer_client::sync() -> bool {
42 auto tus = cbdc::watchtower::tx_id_uhs_ids();
43 for(const auto& [tx_id, tx] : pending_txs()) {
44 auto ctx = transaction::compact_tx(tx);
45 auto [it, success] = tus.insert({ctx.m_id, {}});
46 assert(success);
47 it->second.insert(it->second.end(),
48 ctx.m_inputs.begin(),
49 ctx.m_inputs.end());
50 it->second.insert(it->second.end(),
51 ctx.m_uhs_outputs.begin(),
52 ctx.m_uhs_outputs.end());
53 }
54
55 for(const auto& [tx_id, in] : pending_inputs()) {
56 tus.insert({tx_id, {in.hash()}});
57 }
58
59 cbdc::watchtower::status_update_request req{tus};
60
61 m_logger->debug("Checking watchtower state...");
62
63 auto res = m_wc.request_status_update(req);
64
65 bool success{true};
66 for(const auto& [tx_id, uhs_states] : res->states()) {
67 for(const auto& s : uhs_states) {
68 if((s.status() != cbdc::watchtower::search_status::unspent)
69 && (s.status() != cbdc::watchtower::search_status::spent)) {
70 m_logger->warn("Tx ID:",
71 to_string(tx_id),
72 ", UHS ID:",
73 to_string(s.uhs_id()),
74 ", status:",
75 static_cast<uint32_t>(s.status()));
76 success = false;
77 continue;
78 }
79 break;
80 }
81 if(!success) {
82 continue;
83 }
84 success = confirm_transaction(tx_id);
85 }
86
87 return success;
88 }
